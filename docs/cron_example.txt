# Cron Job Examples for Automated Supabase Backup
# Copy desired line to crontab with: crontab -e

# â­ BACKUP EVERY 5 MINUTES (Recommended)
*/5 * * * * /path/to/supabase_backup_py/run_backup.sh

# Backup daily at 2 AM
0 2 * * * /path/to/supabase_backup_py/run_backup.sh

# Backup every 6 hours
0 */6 * * * /path/to/supabase_backup_py/run_backup.sh

# Backup every Monday at 3 AM
0 3 * * 1 /path/to/supabase_backup_py/run_backup.sh

# Backup every weekday (Monday-Friday) at 1 AM
0 1 * * 1-5 /path/to/supabase_backup_py/run_backup.sh

# Format: minute hour day month weekday command
# */5 = every 5 minutes
# * = every
# */6 = every 6 (hours, days, etc.)
# 1-5 = range (Monday-Friday)
# 0,30 = at minute 0 and 30

# Alternative: Direct Python command
*/5 * * * * cd /path/to/supabase_backup_py && /usr/bin/python3 supabase_backup.py backup >> logs/cron.log 2>&1
